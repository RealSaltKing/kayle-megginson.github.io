Dijkstra's Shortest Path Algorithm¶

Instructions¶

Welcome to the Dijkstra's Shortest Path Finder! This program utilizes Dijkstra's algorithm to find the shortest path between two airports from a provided dataset.

How to Run:¶

  1. Start the Program:

    • Launch the program by running it with Python: python program.py

  2. Input Airport Codes:

    • You will be prompted to enter the IATA airport code for the origin airport.
    • After entering the origin, you will be prompted to enter the IATA airport code for the destination airport.

  3. View the Shortest Path:

    • The program will display the shortest path from the origin to the destination airport along with the total travel distance.

  4. Repeat or Quit:

    • After displaying the shortest path, you can continue to find another path by entering new airport codes.
    • To exit the program, type 'Q' when prompted for the origin airport code.

Limitations:¶

  • Airport Data:

    • The program is currently limited to the airports and routes defined in the provided dataset.
    • Some airports or routes may not be available, which would result in the program indicating that the destination is unreachable.
    • If you want to try different datasets, you will have to go into the code and change which csv file it is supposed to open

  • Dataset Accuracy:

    • The distances are calculated using the Haversine formula based on airport latitude and longitude. As such, the actual travel distance may vary due to real-world factors such as air traffic control restrictions, flight patterns, and more.

  • Performance:

    • While the program is optimized for smaller datasets, performance may degrade with extremely large datasets or complex route networks.

import csv
from typing import Any, NamedTuple
from datastructures.graph import Graph
from haversine import haversine, Unit

class Airport(NamedTuple):
    id: int
    name: str
    code: str
    city: str
    country: str
    latitude: float
    longitude: float

class Flight():
    def __init__(self, origin: Airport, destination: Airport) -> None:
        self._origin: Airport = origin
        self._destination: Airport = destination

    @property
    def weight(self) -> float:
        origin = (self._origin.latitude, self._origin.longitude)
        destination = (self._destination.latitude, self._destination.longitude)
        return haversine(origin, destination, Unit.MILES)

    def __lt__(self, object: Any) -> bool:
        if not isinstance(object, Flight):
            return False

        return self.weight < object.weight


def read_csv(filename: str) -> Graph:
    graph = Graph()

    csv_reader = csv.DictReader(open(filename, newline="\n", encoding="utf-8-sig"))

    for row in csv_reader:
        origin = Airport(
            int(row['origin_airport_id']),
            row['origin_airport'],
            row['origin_airport_code'],
            row['origin_city'],
            row['origin_country'],
            float(row['origin_airport_latitude']),
            float(row['origin_airport_longitude']))

        destination = Airport(
            int(row['destination_airport_id']),
            row['destination_airport'],
            row['destination_airport_code'],
            row['destination_city'],
            row['destination_country'],
            float(row['destination_airport_latitude']),
            float(row['destination_airport_longitude']))

        flight = Flight(origin, destination)
        weight = flight.weight
        graph.add_edge(origin, destination, flight, weight)

    return graph

def dijkstra(graph: Graph, source: Airport, destination: Airport) -> tuple[list, float]:
    distances = {airport: float('inf') for airport in graph.vertices}
    distances[source] = 0
    previous_nodes = {airport: None for airport in graph.vertices}

    unvisited = set(graph.vertices)
    current_vertex = source

    while unvisited:
        current_vertex = min((v for v in unvisited if v in distances), key=lambda x: distances[x], default=None)

        if current_vertex is None or distances[current_vertex] == float('inf'):
            break

        unvisited.remove(current_vertex)

        # Process each edge starting from current_vertex
        for flight in graph.edges_from(current_vertex):
            neighbor = flight._destination
            weight = flight.weight

            if neighbor in unvisited:
                new_distance = distances[current_vertex] + weight
                if new_distance < distances[neighbor]:
                    distances[neighbor] = new_distance
                    previous_nodes[neighbor] = current_vertex

    # Construct shortest path from source to destination
    path = []
    step = destination
    if distances[destination] == float('inf'):
        return [], float('inf')  # Early exit if destination is unreachable

    while step is not None:
        path.append(step)
        step = previous_nodes[step]
    path.reverse()

    return path, distances[destination]



def get_airport_by_input(graph, input_str):
    # Try to match by airport code directly
    matched_airports = [airport for airport in graph.vertices if airport.code.lower() == input_str.lower()]

    if len(matched_airports) == 1:
        return matched_airports[0]
    elif len(matched_airports) > 1:
        print("Multiple airports found:")
        for idx, airport in enumerate(matched_airports, start=1):
            print(f"{idx}. {airport.name} ({airport.code}) - {airport.city}, {airport.country}")
        selection = int(input("Please select the correct airport by number: ")) - 1
        return matched_airports[selection]
    else:
        print("No matching airport found.")
        return None

def main():
    graph = read_csv("routes.csv")

    while True:  # Continuously ask for input until the user decides to quit
        origin_input = input("Enter origin airport code or Q to quit: ")
        if origin_input.lower() == 'q':
            print("See you next flight!")
            break

        destination_input = input("Enter destination airport code: ")

        origin_airport = get_airport_by_input(graph, origin_input)
        destination_airport = get_airport_by_input(graph, destination_input)

        if not origin_airport or not destination_airport:
            print("Origin or destination airport not correctly identified.")
            continue  # Prompt again for input

        shortest_path, total_distance = dijkstra(graph, origin_airport, destination_airport)

        # Output
        print("Shortest path between cities:")
        cumulative_distance = 0.0
        previous_airport = None
        for index, airport in enumerate(shortest_path):
            if index == 0:
                print(f"{airport.code} - {airport.city} (United States) - 0.00 miles")
            else:
                leg_distance = dijkstra(graph, previous_airport, airport)[1]
                cumulative_distance += leg_distance
                print(f"    ✈ {airport.code} - {airport.city} (United States) - {cumulative_distance:.2f} miles")
            previous_airport = airport
        print("Total distance of the shortest path:", f"{total_distance:.2f}", "miles")




if __name__ == "__main__":
    main()
